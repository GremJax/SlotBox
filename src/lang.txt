shape Point {
    X int32
    Y int32
}

shape Box {
    Width int32
    Height int32
}

shape Named {
    Name string
}

object firstThing
firstThing := Point
print firstThing

firstThing := Named
print firstThing

firstThing.X = 100
firstThing.Y = 20
firstThing.Name = "Thing"
print firstThing

firstThing =: Named
print firstThing

firstThing := Box {
    Width -> X
    Height -> Y
}
print firstThing
